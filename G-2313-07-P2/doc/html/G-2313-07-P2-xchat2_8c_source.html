651 long IRCInterface_Connect(char *nick, char *user, char *realname, char *password, char *server, int port, boolean ssl)
652 {
653 
654 
655  struct sockaddr_in serveraddr;
656  struct hostent *servidor;
657  char *msgNick = NULL, *msgUser = NULL, *msgPass = NULL, *msg = NULL;
658  thread_info* info = NULL;
659 
660  if(nick == NULL || user == NULL || realname == NULL || server == NULL)
661  return IRCERR_NOCONNECT;
662 
663  /*Creamos el socket para el cliente*/
664  desc = socket(AF_INET, SOCK_STREAM, 0);
665  if (desc < 0) {
666  syslog(LOG_INFO, "CLIENTE: Error creando el socket");
667  return IRCERR_NOCONNECT;
668  }
669 
670  /*Obtenemos la DNS del servidor*/
671  servidor = gethostbyname(server);
672  if (servidor == NULL) {
673  syslog(LOG_INFO, "CLIENTE: No existe el servidor %s\n", server);
674  return IRCERR_NOCONNECT;
675  }
676 
677  /*Comenzamos la conexion*/
678  bzero((char *) &serveraddr, sizeof (serveraddr));
679  serveraddr.sin_family = AF_INET;
680  bcopy((char *) servidor->h_addr, (char *) &serveraddr.sin_addr.s_addr, servidor->h_length);
681  serveraddr.sin_port = htons(port);
682 
683  /*Conexion con el servidor*/
684  if (connect(desc, (struct sockaddr*) &serveraddr, sizeof (serveraddr)) < 0) {
685  syslog(LOG_INFO, "CLIENTE: Error conectando con el servidor %s", server);
686  return IRCERR_NOCONNECT;
687  }
688 
689  /*Generamos los mensajes para registrar al usuario*/
690  IRCMsg_Nick(&msgNick, NULL, nick, NULL);
691  IRCMsg_User(&msgUser, NULL, user, server, realname);
692  IRCMsg_Pass(&msgPass, NULL, password);
693 
694  /*Juntamos todos los mensajes en uno solo y lo enviamos al servidor*/
695  if(strcmp(password, "") != 0){
696  IRC_PipelineCommands(&msg, msgNick, msgUser, msgPass, NULL);
697  }else{
698  IRC_PipelineCommands(&msg, msgNick, msgUser, NULL);
699  }
700 
701 
702  if (send(desc, msg, strlen(msg), 0) < 0) {
703  syslog(LOG_INFO, "CLIENTE: Error al enviar el mensaje de registro al servidor");
704  return IRCERR_NOCONNECT;
705  }
706  syslog(LOG_INFO, "CLIENTE: Mensaje para registro enviado al servidor %s", server);
707 
708  /*Enviamos al registro plano los mensajes que acabamos de enviar*/
709  IRCInterface_PlaneRegisterOutMessage(msg);
710 
711  /*Liberamos memoria*/
712  IRC_MFree(4, &msg, &msgNick, &msgUser, &msgPass);
713 
714  /*Creamos hilo encargado de PING*/
715  info = (thread_info*) malloc(sizeof (thread_info));
716  info->sd = desc;
717  pthread_create(&thread_ping, NULL, IRC_Client_Ping, (void *) info);
718 
719  /*Creamos el hilo encargado de recibir*/
720  info = (thread_info*) malloc(sizeof (thread_info));
721  info->sd = desc;
722  pthread_create(&thread_recv, NULL, IRC_Client_Recieve, (void *) info);
723 
724  conex.nick = nick;
725  conex.user = user;
726  conex.realname = realname;
727  conex.password = password;
728  conex.server = server;
729  conex.port = port;
730  conex.desc = desc;
731 
732  return IRC_OK;
733 }
